def userinfo(user):
    
    us = df_combine[df_combine['Account'] == user]

    print(f"Min PNL of {user} is ",us['Closed PnL'].min())
    print("Max Size USD : ",user1['Size USD'].max())

    print("Max. traded COIN : ",us['Coin'].value_counts().idxmax())
    print("Trader traded max. trade of classification : ",us['classification'].value_counts().idxmax())
    
    
    return
